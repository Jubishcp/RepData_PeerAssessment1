###Devise a strategy for filling in all of the missing values in the dataset

fillEmpty <- function(steps, interval) {
  retValue <- NA
  if (!is.na(steps))
    retValue <- c(steps)
  else
    retValue <- (stepsinterval[stepsinterval$interval==interval, "steps"])
  return(retValue)
}

##Are there differences in activity patterns between weekdays and weekends? ###Create a new factor variable in the dataset with two levels - “weekday” and “weekend” indicating whether a given date is a weekday or weekend day.

finddaytype <- function(date) 
  {
    day <- weekdays(as.Date(date))
    if (day %in% c("Saturday", "Sunday")) {
        "weekend"
    } else {
        "weekday"
    }
}


filleddataset$date <- as.Date(filleddataset$date)
filleddataset$day <- as.factor(sapply(filleddataset$date, FUN=finddaytype))
